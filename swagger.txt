1、项目属性-生成中，勾选生成xml文档

        var xmlPath = System.IO.Path.Combine(basePath, "Iot.Max.Api.xml"); //Iot.Max.Api.xml为第一步勾选中显示的名字
        c.IncludeXmlComments(xmlPath);
